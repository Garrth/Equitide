    ```python
    def apply_prandtl_ll(CL_2D, CD_0, AR, e):
        """
        C_L_3D = C_L_2D / (1 + 2/AR)          [Anderson §5.3 Eq. 5.61]
        C_D_i  = C_L_3D^2 / (pi * e * AR)     [Eq. 5.62]
        C_D_total = C_D_0 + C_D_i
        """
        CL_3D = CL_2D / (1.0 + 2.0 / AR)
        CD_i = CL_3D**2 / (np.pi * e * AR)
        CD_total = CD_0 + CD_i
        LD_3D = CL_3D / CD_total
        return CL_3D, CD_i, CD_total, LD_3D

    # Verification: at AR=50, CL_3D = CL_2D/(1+2/50) = CL_2D/1.04 → difference ~3.85%
    ```

    Step 5 — Rotating-arm velocity triangle (core geometry; Pitfall fp-chain-loop-geometry guard):

    ```python
    def velocity_triangle(lambda_val, v_loop):
        """
        Rotating-arm geometry (NOT chain-loop):
        v_tangential = lambda * v_loop  [horizontal, tangential to rotation]
        v_loop       = vessel speed     [vertical, upward for ascending]
        v_rel        = sqrt(v_loop^2 + v_tangential^2)
        beta         = arctan(v_loop / v_tangential)  [angle from horizontal toward vertical]
        """
        v_tan = lambda_val * v_loop
        v_rel = np.sqrt(v_loop**2 + v_tan**2)
        beta_rad = np.arctan2(v_loop, v_tan)  # arctan(v_loop/v_tan); = pi/2 as lambda->0
        beta_deg = np.degrees(beta_rad)
        return v_tan, v_rel, beta_rad, beta_deg
    ```

    ```python
    def compute_foil_forces(lambda_val, v_loop, AoA_target_deg, mount_angle_rad, AR, e, A_foil):
        """
        For ascending vessel:
        beta = arctan(v_loop / v_tangential) from horizontal
        F_tan = L*sin(beta) - D*cos(beta)  [drives shaft in rotation direction]
        F_vert = -L*cos(beta) - D*sin(beta) [opposes ascent; negative = upward obstruction]

        L/D minimum for net F_tan > 0:
        F_tan > 0 iff L*sin(beta) > D*cos(beta) iff L/D > cos(beta)/sin(beta) = cot(beta) = lambda
        Note: cot(beta) = v_tangential/v_loop = lambda (when v_tangential = lambda*v_loop exactly)
        Also expressible as: L/D > lambda (for net tangential gain at any lambda)
        Cross-check vs CONTEXT: CONTEXT states (L/D)_min = sqrt(1+1/lambda^2).
        Derivation check: cot(beta) = v_tan/v_loop = lambda.
        sqrt(1+1/lambda^2) = sqrt((lambda^2+1)/lambda^2) = sqrt(1+lambda^2)/lambda = v_rel/v_tan.
        These are different: lambda vs v_rel/v_tan. Report both and identify the correct threshold.
        Correct: (L/D)_min for F_tan > 0 = cot(beta) = lambda.
        The CONTEXT formula v_rel/v_tan = sqrt(1+1/lambda^2) is the ratio v_rel/v_tangential.
        Document this resolution in the output JSON.
        """
        v_tan, v_rel, beta_rad, beta_deg = velocity_triangle(lambda_val, v_loop)

        # AoA effective (Pitfall m2 guard)
        AoA_eff_deg = compute_AoA_effective(beta_rad, mount_angle_rad)

        # Clamp to table range; flag stall
        stall_flag = "OK"
        if AoA_eff_deg > 14:
            stall_flag = "STALL"
        elif AoA_eff_deg > 12:
            stall_flag = "NEAR_STALL"

        # Interpolate NACA table (linear between adjacent entries)
        alpha_clamp = np.clip(AoA_eff_deg, 0, 14)
        CL_2D, CD_0 = interpolate_naca(alpha_clamp, naca_data)

        # 3D corrections
        CL_3D, CD_i, CD_total, LD_3D = apply_prandtl_ll(CL_2D, CD_0, AR, e)

        # Lift and drag (using v_rel)
        q = 0.5 * rho_w * v_rel**2 * A_foil  # dynamic pressure * area [N]
        L = q * CL_3D   # [N]
        D = q * CD_total  # [N]

        # Force decomposition in rotating-arm frame
        F_tan = L * np.sin(beta_rad) - D * np.cos(beta_rad)   # drives shaft
        F_vert = -L * np.cos(beta_rad) - D * np.sin(beta_rad) # opposes ascent (negative = upward force opposing)

        # Minimum L/D for net F_tan > 0
        LD_min = np.cos(beta_rad) / np.sin(beta_rad)  # = cot(beta) = lambda

        # Shaft power
        P_shaft = F_tan * v_tan  # [W]; positive when F_tan > 0

        # Reynolds number
        Re = v_rel * chord / nu_w

        return {
            "lambda": lambda_val, "beta_deg": beta_deg, "v_tangential_ms": v_tan,
            "v_rel_ms": v_rel, "AoA_eff_deg": AoA_eff_deg, "stall_flag": stall_flag,
            "CL_2D": CL_2D, "CL_3D": CL_3D, "CD_0": CD_0, "CD_i": CD_i,
            "CD_total": CD_total, "LD_3D": LD_3D, "L_N": L, "D_N": D,
            "F_tan_N": F_tan, "F_vert_N": F_vert,
            "LD_min_threshold": LD_min, "P_shaft_W": P_shaft, "Re": Re
        }
    ```
